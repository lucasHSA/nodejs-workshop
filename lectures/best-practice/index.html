Singleton-Module

Beispiel

Wir möchten ein Datenbank-Modul schreiben, welches wir an mehreren Stellen requiren möchten. Es soll immer die selbe Datenbank-Handler-Instanz zurückgeben.


Ideen?

Singleton-Module

var dbInstance = null;

function connect(callback) {

    db.connect(function (err, dbHandler) {
        if(err) { callback(err); return; }

        //store instance
        dbInstance = dbHandler;
        callback(null);
    });
}

function getInstance() {
    return dbInstance;
}


exports.connect = connect;
exports.disconnect = disconnect;
exports.getInstance = getInstance;

Anwendung

//app.js
var database = require("./db.js");

//datenbank initialisieren
database.connect(function(err) {

    if(err) { throw err; }

    app.get("/foo", require("./readFoo.js");

    app.listen(3000);
});
//printFoo.js
var db = require("./db.js");

function printFoo(req, res){

    //instance holen, nachdem .connect() aufgerufen wurde
    var dbInstance = db.getInstance();

    dbInstance.get('foo', function (err, value) {
        if (err) { res.send('foo does not exist'); return; }
        res.send("foo => " + value);
    });
}

Singleton - Property

var dbInstance = null;

function connect(callback) {

    db.connect(function (err, dbHandler) {

        dbInstance = dbHandler;

        //instance neu setzen
        //damit von außen erreichbar
        exports.instance = dbInstance;
        callback(null);

}

exports.connect = connect;

Auto-Init (nur sync)

Beispiel: Express Instanz als Singleton


//server.js
var express = require('express');
var app = null;

function getInstance() {

    //works only for sync initialization
    if(app === null) {
        app = express();
    }

    return app;
}

module.exports = getInstance;

Anwendung

//app.js
var server = require("./server.js");

//=getInstance call
//with auto init
var app = server();

app.set("title", "Singletons!");

//server starten
app.listen(3000);
//printAppTitle.js
var server = require("./server.js");

function printAppTitle(req, res) {

    var app = server();
    res.send(app.get("title"));
}

Singletons

  • praktisch um Instanzen zu teilen
  • connect-Methode bei asynchronener Initialisierung nötig
  • connect muss aufgerufen worden sein, bevor auf instance zugegriffen werden kann
  • bei synchroner Initialisierung kann connect in getInstance automatisch passieren

Errors via Events


JSONReader.prototype.fromFile = function(path, callback) {

    var self = this;

    fs.readFile(path, { encoding : "utf-8"}, function(err, jsonStr) {

        if(err) { self.emit("error", err); return; }

        try {
            callback(JSON.parse(jsonStr));
        }
        catch(err) {
            self.emit("error", err);
        }
    });
};

Flow-Control

Beispiel: listDir

Wir möchten eine Funktion listDir(dirname, callback) schreiben. Die Funktion liest das Verzeichnis dirname asynchron ein und ruft anschließend callback(null, results) auf, wobei results ein Object ist, das die Verzeichnisstruktur widerspiegelt. 

{
    'express-app': {
        'package.json': 'file',
        'test.js': 'file'
    },
    'init-modules': {
        routeDefintion: {
            'addRoutes.js': 'file',
            'server.js': 'file'
        },
        singletonProperty: {
            'app.js': 'file',
            'routes.js': 'file'
        }
    }
}

Schritt 1: Verzeichnis einlesen

fs.readdir(dirname, function (err, files) {
    if (err) {
        callback(err);
        return;
    }
});

Schritt 2: fs.stat auf alle Dateien im Verzeichnis ausführen

files.forEach(function (filename) {
    var path = pathUtil.join(dirname, filename);

    fs.stat(path, function (err, stat) {
        if (err) {
            callback(err);
            return;
        }
    });
});

Schritt 3: Rekursion und Ergebnis abspeichern

if (stat.isDirectory()) {
    listDir(path, function (err, partialResult) {
        if (err) {
            callback(err);
            return;
        }

        result[filename] = partialResult;
    });
} else {
    result[filename] = "file";
}

Problem: Wann sind alle fs.stat fertig?

Schritt 4: done()-Funktion einführen und Callbacks zählen

pending = files.length;
function done(filename, partialResult) {
    pending--;
    result[filename] = partialResult;
    if (pending === 0) {
        callback(null, result);
    }
}
done(filename, partialResult);
done(filename, "file");

Welcome to the callback-hell!

  • Asynchroner Code ist schwerer zu lesen
  • Schleifen und Rekursion machen den Control-Flow undurchsichtig
  • Abhängigkeiten von asynchronen Funktionen benötigen zusätzliche Logik (Stichwort: pending)

Deshalb sollte man bei komplizierteren, asynchronen Funktionen eine Control-Flow-Lib verwenden.

Möglichkeit 1: async

https://github.com/caolan/async
npm install async

async.parallel

async.parallel([
    function A() { ... },
    function B() { ... },
    function C() { ... }
], callback);
  • Führt A, B, C gleichzeitig aus
  • Falls eine Funktion mit err antwortet, wird sofort callback(err) aufgerufen
  • Wenn alle Funktionen ohne Fehler geantwortet haben wird callback(null, results) ausgeführt, wobei results die Ergebnisse von A, B, C beinhalten

async.series

async.series([
    function A() { ... },
    function B() { ... },
    function C() { ... }
], callback);
  • Führt erst A aus
  • Falls A mit err antwortet, wird sofort callback(err) aufgerufen
  • Ansonsten wird anschließend B aufgerufen, usw.
  • Wenn alle Funktionen ohne Fehler geantwortet haben wird callback(null, results) ausgeführt, wobei results die Ergebnisse von A, B, C beinhalten

async.waterfall

async.waterfall([
    function A() { ... },
    function B() { ... },
    function C() { ... }
], callback);
  • Wie async.series, aber ...
  • nachdem A geantwortet hat, wird B mit dem Ergebnis von A aufgerufen usw.

async.whilst

var count = 0;

async.whilst(
    function test() { return count < 5; },
    function action(callback) {
        count++;
        setTimeout(callback, 1000);
    },
    function (err) {
        // 5 seconds have passed
    }
);
  • Entspricht einer asynchronen while-Schleife
  • Führt action aus, solange test true returned

Möglichkeit 2: Promises

If a function cannot return a value or throw an exception without blocking, it can return a promise instead. A promise is an object that represents the return value or the thrown exception that the function may eventually provide.
http://documentup.com/kriskowal/q/#

Spezifikation

Promises/A+

  • Ein Promise ist ein Objekt, das eine then()-Methode bereitstellt
  • Ein Promise hat drei Zustände: pending, fulfilled, rejected
  • fulfilled ist das Äquivalent zu return, rejected zu throw
  • Die then()-Methode akzeptiert zwei Funktionen:
    • onFulfilled
    • onRejected
  • then() gibt wiederum ein Promise zurück, dass sich nach dem zurückgegebenen Wert von onFulfilled oder onRejected richtet

Achtung: Namensverwirrung

Promises werden manchmal auch Deferred, Futures oder Thenables genannt. Es gibt verschiedene Implementierungen mit jeweils kleinen Unterschieden. Generell aber gilt:

Ein Promise wird nach "nach außen" abgegeben. Es ermöglicht nur das Ausführen von Funktionen, wenn ein bestimmter Zustand eingetreten ist. Der Zustand selber kann nicht beeinflusst werden.

Ein Deferred ist ein Hilfs-Objekt, das üblicherweise nicht nach außen gegeben werden sollte. Über ein Deferred kann der Zustand eines Promises beeinflusst werden.

Beispiel

Wir möchten fs.readFile() so umbauen, dass es ein Promise zurückgibt. Als Promise-Lib verwenden wir q 

function readFile(file, options) {
    var deferred = Q.defer();

    fs.readFile(file, options, function onFileRead(err, data) {
        if (err) {
            deferred.reject(err);
            return;
        }

        deferred.resolve(data);
    });

    return deferred.promise;
}
readFile("someFile.txt", { encoding: "utf8" })
    .then(function (data) {
        console.log(data);
    });

Oder einfach...

Q.nfcall(fs.readFile, "someFile.txt", { encoding: "utf8" })
    .then(function (data) {
        console.log(data);
    });

Wenn node Promises verwenden würde...

Beispiel: Wir möchten eine Funktion readFiles() schreiben, die ein Array von Dateien akzeptiert, diese asynchron einliest und die jeweiligen Inhalte gesammelt als Array zurückgibt.

Ohne Promises 

function readFiles(files, callback) {
    var allData = [],
        pending = files.length;

    files.forEach(function (file) {
        fs.readFile(file, function onFileRead(err, data) {
            if (err) {
                callback(err);
            }
            pending--;
            done(file, data);
        });
    });

    function done(file, data) {
        allData.push(data);
        if (pending === 0) {
            callback(null, allData);
        }
    }
}

Mit Promises 

function readFiles(files) {
    var promises;

    promises = files.map(fs.readFile);

    return Q.all(promises);
}
