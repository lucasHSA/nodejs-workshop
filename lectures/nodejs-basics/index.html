Modul-Definition

Single Export

    console.log("Hello World!");
}

>require("./helloWorld.js");
[Function: helloWorld]

> var helloWorld = require("./helloWorld.js");
> helloWorld();

Multiple Exports

//hello.js
function hello(who) {
    console.log("hello " + who);
}

function helloWorld() { hello("world"); }

function helloSun() { hello("sun"); }

exports.helloWorld = helloWorld;
exports.helloSun = helloSun;

> require("./hello.js");
{ helloWorld: [Function: helloWorld],
helloSun: [Function: helloSun] }
> hello.hello
undefined

export / exports

  • export für mehrere "exports"
  • module.exports für einzelnen "export"
  • Funktionen die nicht exportiert werden sind nur im Module-Scope sichtbar

Require Directory

//folderModule/m1.js
module.exports = "m1";

//folderModule/m2.js
module.exports = "m2";

//folderModule/index.js
exports.m1 = require("./m1.js");

> require("./folderModule");
{ m1: 'm1', m2: 'm2' }

  • Beim 'requiren' eines Ordners wird die index.js geladen (falls vorhanden)
  • sehr praktisch um kleinere Module zusammenzufassen

getter vs. properties

var mood = "grumpy";

function get() {
    return mood;
}

function set(newMood) {
    mood = newMood;
}

exports.get = get;
exports.set = set;
//erzeugt eine Kopie des String
//keine Referenz zu var mood
exports.mood = mood;


> mood.get(); //=grumpy
> mood.mood = "happy";
> mood.get(); //=grumpy


> mood.set("happy");
> mood.get(); //=happy

Zusammenfasssung

  • Module werden nach dem ersten Laden gecached
  • jeder require-Aufruf liefert danach dasselbe Objekt zurück
  • über Funktionen kann man Variablen innerhalb der Module beeinflussen
  • Attribute sollten nur als Konstanten genutzt werden

Events

  • require("events");
  • Kernkomponente von Node.js
  • wichtig für asynchrone Systeme
  • siehe JavaScript im Browser (onClick, etc.)
var EventEmitter = require("events").EventEmitter;

//Monster erbt von EventEmitter ab
util.inherits(Monster, EventEmitter);

Monster.prototype.growl = function() {
    //wir triggern das event "growl"
    this.emit("growl", "BOOOAARRR");
};
var giantSquirrel = new Monster("GiantSquirrel");

//mit "on" hören auf das event "growl"
giantSquirrel.on("growl", function(growl) {
    console.log(this.name + " growled: " + growl);
});

giantSquirrel.growl();

Streams

  • require("stream");
  • Kernkonzept von Node.js
  • Readable Streams
  • Writeable Streams
  • Duplex Streams
  • z.B. HTTP / FS

Readable Stream

var fs = require("fs");

var file = "../fs/files/names.json";

var namesStream = fs.createReadStream(file, { encoding : "utf-8" });

namesStream.on("readable", function() {
    console.log(namesStream.read());
});

namesStream.on("end", function() {
    console.log("end");
});

namesStream.on("error", function(e) {
    throw e;
});

Writeable Stream

function getDate() {
    return new Date().toISOString();
}

function printDate() {
    setTimeout(function() {
        //process.stdout = writeable stream
        process.stdout.write("It is: " + getDate() + "\n", printDate);
    }, 1000);
}

printDate();

Readable <> Writeable

http.createServer(function (req, res) {

    var names = fs.createReadStream("./data.json", { encoding : "utf-8" });

    //names = readable stream
    names.on("readable", function() {
        //res = writeable stream
        res.write(names.read());
    });

    names.on("end", res.end);

}).listen(9000, "127.0.0.1");

Stream Pipe

http.createServer(function (req, res) {

    var names = fs.createReadStream("./data.json", { encoding : "utf-8" });

    //ersetzt das manuelle weitergeben
    //wie | in der shell
    names.pipe(res);
});

Streams

  • Readable Stream | Writeable Stream
  • Writeable Stream | Readable Stream
  • Readable: Http.IncomingMessage, process.stdin, fs.ReadStream
  • Writeable: Http.ServerResponse, process.stdout, fs.WriteStream
  • Daten müssen nicht komplett in den Arbeitsspeicher
  • partielles Bearbeiten von Daten

Eigene Module

Module

  • bezeichnet eine Bibliothek
  • hat eine bestimmte Struktur
  • wird durch die package.json definiert

Struktur

  • ./lib die eigentliche Bibliothek
  • ./test tests (falls vorhanden)
  • ./node_modules enthält alle Module von denen das Modul abhängt
  • package.json definiert das Modul
  • README.md Beschreibung d. Moduls
  • LICENSE Lizenzbedingungen

package.json

{
    "name": "myModule",
    "version": "0.0.1",
    "description": "example module",
    "main": "./lib/index.js",
    "scripts": {
        "test": "mocha ./test"
    },
    "dependencies" : {
        "connect" : "2.9.x"
    },
    "devDependencies" : {},
    "repository": "",
    "author": "meaku",
    "license": "MIT"
    }

package.json



//package.json
{
    "name" : "myModule",
    "main" : "./lib/index.js"
}
//lib/index.js
exports.sayHello = function() {
    console.log("hello module");
};

var myModule = require("myModule");
myModule.sayHello();
//> hello module
